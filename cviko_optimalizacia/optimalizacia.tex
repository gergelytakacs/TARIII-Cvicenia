\begin{eqnarray*}

\begin{frame}
Zovšeobecnené riešenie tohto problému musí spåòa podmienky KKT (Karush-Kuhn-Tucker)\\ \vspace{0.4cm}
Nutné podmienky
\begin{itemize}
\item 1. Stacionarita (Stationarity)\\
- pre max.: $\nabla f(x^\star) = \sum_{i=1}^{m}\mu_i\nabla g_i(x^\star) + \sum_{i=1}^{l}\lambda_j\nabla h_j(x^\star)$ \\
- pre min.:  $-\nabla f(x^\star) = \sum_{i=1}^{m}\mu_i\nabla g_i(x^\star) + \sum_{i=1}^{l}\lambda_j\nabla h_j(x^\star)$
\item 2. Primárna prípustnos (Primal feasibility)\\
-  $g_i(x^\star) \leq 0$, pre všetky $i = 1,...,m$\\
-  $h_j(x^\star) = 0$, pre všetky $j = 1,...,l$
\item 3. Duálna prípustnos (Dual feasibility)\\
- $\mu_i \geq 0$, pre všetky $i = 1,...,m$
\item 4. Doplnková vô¾a (Complementary slackness)\\
- $\mu_ig_i(x^\star) = 0$, pre všetky $i = 1,...,m$
\end{itemize}
\vspace{0.4cm}
- $x^\star$ oznaèuje polohu optima
\end{frame}
